# ---------------------------------------------------------------------------
# Audio processing
# ---------------------------------------------------------------------------
# librosa — kept despite torchaudio overlap for three specific uses:
#   1. librosa.filters.mel() in mel_processing.py + speaker_encoder.py
#      (must match NeMo's exact filterbank; torchaudio.functional.melscale_fbanks
#       uses a different normalisation that would break loaded checkpoints)
#   2. librosa.feature.rms() in pipeline.py (protect-consonants logic)
#   3. librosa.resample() in onnx/infer.py + voice_analysis.py
#   All three are shallow calls; the rest of librosa is unused.
#   Replacing them with torchaudio/scipy equivalents is possible but risky —
#   checkpoint-trained weights depend on exact filterbank values.
librosa
soundfile                 # sf.write / sf.read used throughout
sounddevice               # realtime mic/speaker I/O
av                        # audio I/O in rvc/infer/lib/audio.py; bundles its own FFmpeg — no system ffmpeg needed
noisereduce               # audio_preprocessing.py noise removal
praat-parselmouth         # F0 extraction via PM method in rvc/rvc/f0/pm.py
pyworld==0.3.2            # pinned — API changed in 0.3.3+; DIO/Harvest F0

# REMOVED: pydub — was used only for MP3/M4A output in offline router.
#   Offline inference now forces WAV output (out_ext=".wav"), so pydub is dead code.
#   Also removes the system ffmpeg dependency (av bundles its own).
# REMOVED: resampy — not imported anywhere in the codebase; was a transitive dep of librosa
#   but pip resolves it automatically if librosa needs it.
# REMOVED: audioread — same story; librosa pulls it in transitively if needed.
